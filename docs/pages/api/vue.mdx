```vue [example.vue]
<script lang="ts" setup>
  import { useStore } from "exome/vue";
  import { counterStore } from "./counter.store.ts";

  const { count, increment } = useStore(counterStore);
</script>

<template>
  <button @click="increment()">{{ count }}</button>
</template>
```

```ts twoslash [counter.store.ts] filename="counter.store.ts"
import { Exome } from "exome"

class CounterStore extends Exome {
  public count = 0

  public increment() {
    this.count += 1
  }
}

export const counterStore = new CounterStore()
```
