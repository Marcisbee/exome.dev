```tsx twoslash [example.tsx]
// @noErrors
// ---cut---
import { useStore } from "exome/solid"
import { counterStore } from "./counter.store.ts"

function App() {
  const count = useStore(counterStore, s => s.count)

  return (
    <button onClick={counterStore.increment}>{count}</button>
  );
}
```

```ts twoslash [counter.store.ts] filename="counter.store.ts"
import { Exome } from "exome"

class CounterStore extends Exome {
  public count = 0

  public increment() {
    this.count += 1
  }
}

export const counterStore = new CounterStore()
```
